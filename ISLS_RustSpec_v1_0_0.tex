\section{Open Issue Resolutions}
\label{sec:oi}
% ═══════════════════════════════════════════════════════════════════════════════

Every open issue from the ISLS Open Issues Register is resolved below. These decisions are \textbf{normative} for this implementation.

\begin{resolvebox}[OI-01 RESOLVED: Canonical Serialization Grammar]
\textbf{Decision:} RFC~8785 JSON Canonicalization Scheme (JCS).\\
\textbf{Rationale:} Human-readable, deterministic, widely implemented, sufficient performance for single-operator system. The crate \texttt{serde\_jcs} (or manual implementation of RFC~8785 rules atop \texttt{serde\_json}) SHALL be used.\\
\textbf{Rules:} Lexicographic key sort by UTF-8 bytes. Numbers: IEEE~754 double-precision serialized per ECMAScript \texttt{Number} rules (no trailing zeros, no \texttt{+} sign). No whitespace. No BOM. UTF-8 only.\\
\textbf{Hash:} SHA-256 over the canonical JCS byte string. The result is a 32-byte \ty{Hash256} = \texttt{[u8; 32]}.
\end{resolvebox}

\begin{resolvebox}[OI-02 RESOLVED: Operator Alphabet and Typing]
\textbf{Decision:} Define a core trait \ty{Operator} with associated types, plus 8 reference implementations.\\
\textbf{Type system:}
\begin{lstlisting}[style=rust]
pub trait Operator: Send + Sync {
    fn id(&self) -> &str;
    fn version(&self) -> &str;
    fn evaluate(&self, input: &FiveDState, params: &ParamMap) -> f64; // -> [0,1]
    fn is_deterministic(&self) -> bool { true }

\textbf{Reference operators:} \texttt{BandOp}, \texttt{RatioOp}, \texttt{CorrelationOp}, \texttt{GrangerOp}, \texttt{SpectralOp}, \texttt{TopologicalOp}, \texttt{PhaseOp}, \texttt{ContractionOp}. Each has fixed type signature, parameter schema, and at least 3 test vectors.
\end{resolvebox}

\begin{resolvebox}[OI-03 RESOLVED: Topological Signature Schema]

\begin{resolvebox}[OI-05 RESOLVED: Proof-Object Verifier Algorithm]
\textbf{Decision:} Deterministic function \func{verify\_crystal} returns \ty{VerifyResult}.
\begin{lstlisting}[style=rust]
pub fn verify_crystal(crystal: &SemanticCrystal, archive: &Archive) -> VerifyResult {
    // 1. Hash-chain integrity
    for (i, entry) in crystal.evidence_chain.iter().enumerate() {
        let recomputed = sha256_jcs(&entry.content);
        if recomputed != entry.digest { return VerifyResult::Fail("hash mismatch"); }
        if i > 0 && entry.prev != Some(crystal.evidence_chain[i-1].digest) {
            return VerifyResult::Fail("chain break");
        }
    }
    // 2. Operator-version match
    for (name, ver) in &crystal.commit_proof.operator_stack {
        if archive.pinned_version(name) != Some(ver) {
            return VerifyResult::Fail("operator drift");
        }
    }
    // 3. Gate satisfaction: all 8 gates >= threshold
    // 4. Dual consensus: primal >= theta AND dual >= theta AND mci >= eta
    // 5. PoR trace: monotonic timestamps, T_min/T_stable/T_latch satisfied
    // 6. Content address: crystal.crystal_id == sha256_jcs(canonical(crystal))
    VerifyResult::Accept
}
\end{lstlisting}
\end{resolvebox}

\begin{resolvebox}[OI-06 RESOLVED: Temperature Calibration]
\textbf{Decision:} $T = c_T \cdot \sigma_{\text{realized}}(W)$ where $\sigma_{\text{realized}}$ is the realized standard deviation of the resonance field over window $W$, and $c_T > 0$ is a configurable scaling constant (default: $c_T = 5.0$).\\
\textbf{Fallback:} If no domain data is available, $T = T_{\text{default}} = 1.0$.\\
\textbf{Regime mapping:} $T < 0.5$: calm. $0.5 \leq T < 2.0$: normal. $T \geq 2.0$: volatile.
\end{resolvebox}

\begin{resolvebox}[OI-07 RESOLVED: Radial Coupling Function]
\textbf{Decision:} $g(r) = \exp(-\mu_r \cdot r^2)$ with $\mu_r > 0$ configurable (default: $\mu_r = 1.0$).\\
\textbf{Verification:} $g(0) = 1$, $g(r) \to 0$ as $r \to \infty$, $g$ is smooth and monotone decreasing for $r \geq 0$. Mandorla coupling $\kappa(t) = \exp(-\lambda \cdot \delta\varphi(t)) \cdot \exp(-\mu_r \cdot r(t)^2) \in [0,1]$.
\end{resolvebox}

\begin{resolvebox}[OI-08 RESOLVED: Intrinsic Flow Field]
\textbf{Decision:} Gradient flow on resonance potential.
\[
  \Phi(\mathbf{H}) = -\nabla V(\mathbf{H}), \quad V(\mathbf{H}) = \tfrac{1}{2}\|\mathbf{H} - \mathbf{H}^*\|^2 + \sum_i \lambda_i \cdot g_i(\mathbf{H})
\]
where $\mathbf{H}^*$ is the current attractor centroid (centroid of the top-$k$ resonant points in the HDAG), $g_i$ are active constraint penalty terms, and $\lambda_i > 0$ are Lagrange multipliers from the constraint program.\\
\textbf{Discretization:} Forward Euler with step $\Delta t_2$: $\mathbf{H}_{n+1} = \mathbf{H}_n + \Delta t_2 \cdot \Phi(\mathbf{H}_n)$.\\
\textbf{Dissipation:} Add damping $-\gamma \mathbf{H}$ with $\gamma > 0$ (default: $\gamma = 0.01$).
\end{resolvebox}

    pub rho: f64,  // density

impl FiveDState {
    pub fn as_array(&self) -> [f64; 5] { [self.p, self.rho, self.omega, self.chi, self.eta] }
    pub fn norm_sq(&self) -> f64 { self.as_array().iter().map(|x| x*x).sum() }
    pub fn distance(&self, other: &Self) -> f64 {
        self.as_array().iter().zip(other.as_array().iter())

}

/// Inverse Weaving: recover emergent constraint program (ISLS Algo 1)
pub fn inverse_weave(
    graph: &PersistentGraph,
    window: &TimeWindow,
    library: &[Box<dyn Operator>],
    config: &ExtractionConfig,
) -> (ConstraintProgram, Vec<VertexId>) // (infogenome, condensed region K*)
{
    let point_cloud = graph.point_cloud_over(window);
    let mut active: Vec<ConstraintCandidate> = library.iter()
        .filter_map(|op| scan_constraint(op, &point_cloud, config))
        .filter(|c| c.coverage >= config.alpha_min)
        .collect();

    let mut program = ConstraintProgram::new();
    let mut region = point_cloud.clone();

    while !active.is_empty() && variance(&region) > config.convergence_tau {
        // Select max info-gain per risk: q(g) = (H(K) - H(K cap g)) / (risk(g) + eps)
        let best = select_best_constraint(&active, &region, config);
        region.retain(|v| best.evaluate(v) >= best.threshold);
        let contraction = region.len() as f64 / point_cloud.len() as f64;
        program.push(best.clone());
        active.retain(|c| c.id != best.id);
        if contraction > config.kappa_max { break; }
    }
    (program, region.into_vertex_ids())
}

/// Lattice free energy: F(L) = -sum(q(g_i)) + T * H(K*) (ISLS Def 14.4, OI-06)
pub fn lattice_free_energy(program: &ConstraintProgram, region: &[FiveDState],
    temperature: f64) -> f64
{
    let info_gain: f64 = program.iter().map(|c| -c.formation_energy).sum();
    let entropy = region_entropy(region);
    -info_gain + temperature * entropy
}

    lock_tick: Option<f64>,

impl PoRFsm {
    pub fn step(&mut self, kappa: f64, t2: f64, config: &ConsensusConfig) -> PoRState {
        match self.state {
            PoRState::Search => {
                if kappa >= config.por_kappa_bar {
                    self.stability_history.push(kappa);
                    if self.stability_history.len() >= config.por_t_min {
                        self.state = PoRState::Lock;
                        self.lock_tick = Some(t2);
                        self.trace.lock_enter = Some(t2);
                    }
                } else { self.stability_history.clear(); }
            }
            PoRState::Lock => {
                let delta = (kappa - self.stability_history.last().unwrap_or(&0.0)).abs();
                if delta <= config.por_epsilon {
                    self.stability_history.push(kappa);
                    if self.stability_history.len() >= config.por_t_min + config.por_t_stable {
                        self.state = PoRState::Verify;
                        self.verify_tick = Some(t2);
                        self.trace.verify_enter = Some(t2);
                    }
                } else { self.reset(t2); }
            }
            PoRState::Verify => { /* check policy constraints, latch time */ }
            PoRState::Commit => { /* terminal until reset */ }
        }
        self.state.clone()
    }
    fn reset(&mut self, t2: f64) {
        self.state = PoRState::Search;
        self.stability_history.clear();
        self.trace = PoRTrace { search_enter: t2, ..Default::default() };
    }
}

/// Verify crystal integrity (OI-05 resolved)
pub fn verify_crystal(crystal: &SemanticCrystal, pinned: &BTreeMap<String,String>)
    -> Result<(), VerifyError>
{
    // 1. Content address
    let recomputed_id = content_address(crystal);
    if recomputed_id != crystal.crystal_id { return Err(VerifyError::ContentAddress); }

    // 2. Hash-chain integrity
    for (i, entry) in crystal.evidence_chain.iter().enumerate() {
        if content_address_raw(&entry.content) != entry.digest {
            return Err(VerifyError::HashChain(i));
        }
        if i > 0 && entry.prev != Some(crystal.evidence_chain[i-1].digest) {
            return Err(VerifyError::ChainBreak(i));
        }
    }

    // 3. Operator versions
    for (name, ver) in &crystal.commit_proof.operator_stack {
        if pinned.get(name) != Some(ver) { return Err(VerifyError::OperatorDrift(name.clone())); }
    }

    // 4. Gate satisfaction (all metrics >= thresholds checked in GateSnapshot.kairos)
    if !crystal.commit_proof.gate_values.kairos { return Err(VerifyError::GateFail); }

    // 5. Dual consensus
    let cr = &crystal.commit_proof.consensus_result;
    if cr.primal_score < cr.threshold || cr.dual_score < cr.threshold || cr.mci < 0.8 {
        return Err(VerifyError::ConsensusFail);
    }

    // 6. PoR trace monotonicity
    // (verify search <= lock <= verify <= commit timestamps)
    Ok(())
}

        || metrics.s_shock >= config.thresholds.s_shock

        attempt_carrier_migration(state, &metrics, config);

    if !gate.kairos {

    let window = TimeWindow::last_n(config.extraction.window_hours);
    let (program, region) = inverse_weave(&state.graph, &window,
        &config.extraction.operators, &config.extraction);

    // Operators: DK + Funnel
    state.engine_state = EngineState::Projecting;
    let condensed = apply_operator_cascade(&state.h5_state, &mandorla, config);

    // Stitching
    state.engine_state = EngineState::Stitching;
    let seam = stitch(&condensed, &region, config);
    if metrics.n_seam < config.thresholds.n {
        state.engine_state = EngineState::Rejected("seam failed".into());
        return Ok(None);

        &config.consensus.primal_ops(), &config.consensus.dual_ops(), &config.consensus);
    if consensus.primal_score < consensus.threshold
        || consensus.dual_score < consensus.threshold
        || consensus.mci < config.consensus.mirror_consistency_eta

    state.engine_state = EngineState::Restoring;

    state.engine_state = EngineState::Idle;

    Ok(Some(crystal))
}
